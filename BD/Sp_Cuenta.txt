-- CUENTA
delimiter //
use CHAT;
//

drop procedure if exists sp_insertar_cuenta;
//
drop procedure if exists sp_actualizar_cuenta;
//
drop procedure if exists sp_buscar_cuenta;
//
drop procedure if exists sp_mostrar_cuentas;
//

drop view if EXISTS V_cuentas;
//

CREATE PROCEDURE sp_insertar_cuenta 
(
    _correo varchar(50),
    _contrasenia varchar(50)
) 
BEGIN
	INSERT INTO CUENTA(correo,contrasenia) VALUES(_correo,_contrasenia);
END
//
CREATE PROCEDURE sp_actualizar_cuenta 
(
    _id_cuenta int,
    _correo varchar(50),
    _contrasenia varchar(50)
) 
BEGIN
	update CUENTA set correo=_correo,contrasenia=_contrasenia where id_cuenta=_id_cuenta;
END
//
CREATE PROCEDURE sp_buscar_cuenta
(
    _id_cuenta int
) 
BEGIN
	Select * from CUENTA where id_cuenta=_id_cuenta;
END
//
CREATE PROCEDURE sp_mostrar_cuentas()
BEGIN
	Select * from CUENTA;
END
//
